  mutate(
    adj_date = case_when(
      date >= as.Date("2000-01-01") & date <= as.Date("2001-12-31") ~ date + years(11),
      date >= as.Date("2007-01-01") & date <= as.Date("2009-12-31") ~ date + years(7),
